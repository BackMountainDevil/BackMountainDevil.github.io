# 查找py
$ whereis python
python: /usr/bin/python3.8 /usr/lib/python3.8 /usr/lib/python2.7 /usr/lib/python3.9 /etc/python3.8 /usr/local/lib/python3.8 /usr/include/python3.8
# 无usr/bin/python，不需要删除
# 建立软连接
$ sudo ln -s /usr/bin/python3.8 /usr/bin/python
# 把路径/usr/bin/加入环境变量PATH中
$ PATH=/usr/bin:$PATH

这个时候在终端敲入python就是进入py3的环境了
